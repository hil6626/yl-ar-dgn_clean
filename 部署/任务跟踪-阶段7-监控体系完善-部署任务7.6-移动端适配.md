```bash

/* 移动端适配样式 */

@media (max-width: 768px) {
    /* 布局调整 */
    .grid-cols-4 {
        grid-template-columns: repeat(2, 1fr);
```

### 步骤2: 创建移动端检测

```bash
cat > /tmp/verify_mobile.sh << 'EOF'
#!/bin/bash
echo "========================================"
echo "移动端适配验证"
echo "========================================"

errors=0

# 检查移动端样式
echo -n "检查移动端样式... "
if [ -f "/home/vboxuser/桌面/项目部署/项目1/yl-ar-dgn_clean/YL-monitor/static/css/mobile.css" ]; then
    echo "✓ 通过"
else
    echo "✗ 失败"
    errors=$((errors + 1))
fi

echo "========================================"
if [ $errors -eq 0 ]; then
    echo "✓ 移动端适配验证通过"
    exit 0
else
    echo "✗ 移动端适配验证失败 ($errors 个错误)"
    exit 1
fi
EOF

chmod +x /tmp/verify_mobile.sh
/tmp/verify_mobile.sh
```
